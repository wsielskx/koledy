% Change the leftbar width
\renewenvironment{leftbar}[1][\hsize]
{%
\def\FrameCommand
{%
    {\vrule width 2pt}
    \hspace{2pt}
    \fboxsep=\FrameSep
}%
\MakeFramed{\hsize#1\advance\hsize-\width\FrameRestore}%
}
{\endMakeFramed}

  \fi}
